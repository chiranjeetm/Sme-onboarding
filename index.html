// ── LIVE NETWORK DRIVER CONNECTIONS ── async function runOCR(context, inputEl) { const file = inputEl.files[0]; if (!file) return; showOCRState(context, 'loading'); const formData = new FormData(); formData.append("file", file); try { // Connects directly to your verified, active Render server instance const response = await fetch(`https://ocr-backend-3hzz.onrender.com/extract/${context}`, { method: 'POST', body: formData }); if (!response.ok) throw new Error(`Server returned status layout: ${response.status}`); const parsedData = await response.json(); fillOCRFields(context, parsedData); showOCRState(context, 'done'); } catch(err) { console.error('OCR Fatal Link Exception:', err); showOCRState(context, 'error'); } } function showOCRState(context, state) { // Matches the exact CSS progress IDs inside your HTML document layout const p = document.getElementById(`ocr-${context}-prog`); const d = document.getElementById(`ocr-${context}-done`); const e = document.getElementById(`ocr-${context}-err`); if(!p || !d || !e) return; p.classList.remove('show'); d.classList.remove('show'); e.classList.remove('show'); if(state === 'loading') p.classList.add('show'); if(state === 'done') d.classList.add('show'); if(state === 'error') e.classList.add('show'); } function fillOCRFields(context, data) { if (!data) return; if (context === 'tl') { document.getElementById('co-name-en').value = data.name_english || ''; document.getElementById('co-name-ar').value = data.name_arabic || ''; document.getElementById('co-tl-no').value = data.licence_number || ''; document.getElementById('co-emirate').value = data.emirate || 'Dubai'; document.getElementById('co-expiry').value = data.expiry_date || ''; if(data.establishment_date) document.getElementById('co-estab').value = data.establishment_date; document.getElementById('co-name-en').classList.add('ocr-filled'); document.getElementById('co-name-ar').classList.add('ocr-filled'); document.getElementById('co-tl-no').classList.add('ocr-filled'); } if (context === 'eid') { document.getElementById('sh-name-en').value = data.name_english || ''; document.getElementById('sh-name-ar').value = data.name_arabic || ''; document.getElementById('sh-nat').value = data.nationality || data.nationality_code || ''; document.getElementById('sh-eid-exp').value = data.expiry_date || ''; document.getElementById('sh-name-en').classList.add('ocr-filled'); } } function goTo(n) { document.querySelectorAll('.screen').forEach(s => s.classList.remove('active')); document.querySelectorAll('.sb-item').forEach(s => s.classList.remove('active')); document.querySelectorAll('.step').forEach(s => s.classList.remove('active')); document.getElementById(`sc-${n}`).classList.add('active'); document.getElementById(`sb-${n}`).classList.add('active'); document.getElementById(`ps-${n}`).classList.add('active'); if(n === 4) renderAuditDocs(); if(n === 5) renderSummaryData(); } function enableManualMode() { document.querySelectorAll('.fi').forEach(el => { el.readOnly = false; el.classList.remove('ocr-filled'); }); } function renderAuditDocs() { const coName = document.getElementById('co-name-en').value || "Applicant Company"; const shName = document.getElementById('sh-name-en').value || "Primary Shareholder"; document.getElementById('docs-list-view').innerHTML = `
📄 1.1 Valid Trade Licence Tree — ${coName}✓ AUTOMATICALLY LINKED
📄 1.2 12-Month Bank Statement Statements
📄 1.3 Partner Identification EID Card — ${shName}Optional Layer
`; } function renderSummaryData() { document.getElementById('review-panel-summary').innerHTML = ` Company Profile Name: ${document.getElementById('co-name-en').value || 'Not specified'}
Trade Registration Code: ${document.getElementById('co-tl-no').value || 'Not specified'}
Target Emirate Domicile: ${document.getElementById('co-emirate').value || 'Dubai'}
Underwriter Requested Limit: AED ${parseFloat(document.getElementById('loan-limit').value || 500000).toLocaleString()} over ${document.getElementById('loan-tenor').value || 36} Months
Risk Profile Verification Status: 100% Validated Engine Parameters. `; } function collectData() { return { company: { nameEn: document.getElementById('co-name-en').value, tlNo: document.getElementById('co-tl-no').value, domicile: document.getElementById('co-emirate').value }, credit: { requestedAmt: document.getElementById('loan-limit').value, termMonths: document.getElementById('loan-tenor').value }, timestamp: new Date().toISOString() }; } function saveDraft() { localStorage.setItem('deem_sme_underwriting_draft', JSON.stringify(collectData())); alert('Draft configuration locked into local security cache.'); } function exportJSON() { const b = new Blob([JSON.stringify(collectData(), null, 2)], {type: 'application/json'}); const a = document.createElement('a'); a.href = URL.createObjectURL(b); a.download = `deem_underwriting_capture_${Date.now()}.json`; a.click(); }
